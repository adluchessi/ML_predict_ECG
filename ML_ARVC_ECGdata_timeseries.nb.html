outliers <- function (x) {
  Q1 <- quantile(x, probs =.05)
  Q3 <- quantile(x, probs =.95)
  iqr = Q3-Q1
 upper_limit = Q3 + (iqr*1.5)
 lower_limit = Q1 - (iqr*1.5)
 x > upper_limit | x < lower_limit
}

remove_outliers <- function (df, cols = names (df[8:181])) {
  for (col in cols) {
    df <- df[!outliers(df[[col]]),]
  }
  df
}

draw_boxplot <- function(df){ 
  df %>%  
    pivot_longer(8:15, names_to="attributes") %>%  
    ggplot(aes(attributes, value, fill=attributes)) + 
    geom_boxplot() 
}
